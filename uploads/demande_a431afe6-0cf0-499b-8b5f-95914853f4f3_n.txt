import React, { useEffect, useMemo, useState } from "react";
import {
  IonPage, IonContent, IonButton, IonIcon, IonToast,
  IonModal, IonInput, IonItem, IonLabel, IonSelect, IonSelectOption,
  IonSegment, IonSegmentButton, IonSearchbar, IonFab, IonFabButton
} from "@ionic/react";
import { addOutline, closeOutline } from "ionicons/icons";
import "./rappels.css";
import { createRappel, getRappels, Rappel, RappelPayload, RappelType } from "../services/rappels";
import TopMenu from "../components/TopMenu";
const todayMidnight = () => {
  const n = new Date();
  return new Date(n.getFullYear(), n.getMonth(), n.getDate());
};
const isOverdue = (iso: string) => {
  if (!iso) return false;
  const d = new Date(iso + "T00:00:00");
  return d < todayMidnight();
};

const Reminders: React.FC = () => {
  const [items, setItems] = useState<Rappel[]>([]);
  const [type, setType] = useState<RappelType>("COMMANDE");
  const [q, setQ] = useState("");

  // modal details
  const [openDetails, setOpenDetails] = useState(false);
  const [selected, setSelected] = useState<Rappel | null>(null);

  // modal add
  const [openAdd, setOpenAdd] = useState(false);

  const [toastOpen, setToastOpen] = useState(false);
  const [toastMsg, setToastMsg] = useState("");

  const [newR, setNewR] = useState<RappelPayload>({
    type: "COMMANDE",
    reference: "",
    description: "",
    datePrevue: "",
    statut: "A_VENIR",
    priorite: "MOYENNE",
    responsable: "",
    // champs optionnels
    client: "",
    machine: "",
    maintenanceType: "PREVENTIVE",
  });

  const refresh = async () => {
    const data = await getRappels();
    setItems(data);

    const lateCount = data.filter(r => r.statut !== "LIVREE" && isOverdue(r.datePrevue)).length;
    if (lateCount > 0) {
      setToastMsg(`⚠️ ${lateCount} rappel(s) en retard`);
      setToastOpen(true);
    }
  };

  useEffect(() => {
    refresh().catch(console.error);
  }, []);

  // KPI
  const lateCount = useMemo(
    () => items.filter(r => r.statut !== "LIVREE" && isOverdue(r.datePrevue)).length,
    [items]
  );

  // List filtered by segment + search
  const list = useMemo(() => {
    const base = items.filter(r => r.type === type);
    const s = q.trim().toLowerCase();
    if (!s) return base;
    return base.filter(r =>
      (r.reference || "").toLowerCase().includes(s) ||
      (r.description || "").toLowerCase().includes(s) ||
      (r.responsable || "").toLowerCase().includes(s) ||
      (r.client || "").toLowerCase().includes(s) ||
      (r.machine || "").toLowerCase().includes(s)
    );
  }, [items, type, q]);

  const openItem = (r: Rappel) => {
    setSelected(r);
    setOpenDetails(true);
  };

  const resetAddForm = (t: RappelType) => {
    setNewR({
      type: t,
      reference: "",
      description: "",
      datePrevue: "",
      statut: "A_VENIR",
      priorite: "MOYENNE",
      responsable: "",
      client: "",
      machine: "",
      maintenanceType: "PREVENTIVE",
    });
  };

  const submitAdd = async () => {
    if (!newR.reference.trim()) return setToastMsg("Référence obligatoire"), setToastOpen(true);
    if (!newR.datePrevue) return setToastMsg("Date prévue obligatoire"), setToastOpen(true);

    // nettoyage minimal
    const payload: RappelPayload = {
      ...newR,
      reference: newR.reference.trim(),
      description: (newR.description || "").trim(),
      responsable: (newR.responsable || "").trim(),
      client: (newR.client || "").trim(),
      machine: (newR.machine || "").trim(),
    };

    await createRappel(payload);
    setToastMsg("Rappel ajouté ✅");
    setToastOpen(true);
    setOpenAdd(false);
    await refresh();
  };

  return (
    <IonPage>
      <IonContent fullscreen className="login-bg">
        
  <TopMenu title="🔔 Rappels & Notifications" />
  <div className="login-shell">
        <div className="login-shell">
          <div className="reminders-card">

            {/* HEADER same style */}
            <div className="form-head">
              <div className="head-row">
             <h3 className="form-title">Suivi des commandes, maintenance et actions importantes</h3>
                <span className="kpi-badge">⚠️ {lateCount}</span>
              </div>
              
            </div>

            {/* SEGMENT */}
            <IonSegment value={type} onIonChange={(e) => setType(e.detail.value as RappelType)} className="seg">
              <IonSegmentButton value="COMMANDE">Commandes</IonSegmentButton>
              <IonSegmentButton value="MAINTENANCE">Maintenance</IonSegmentButton>
            </IonSegment>

            {/* SEARCH */}
            <IonSearchbar
              value={q}
              onIonInput={(e) => setQ(String(e.detail.value ?? ""))}
              placeholder={type === "COMMANDE" ? "Rechercher commande, client..." : "Rechercher machine, technicien..."}
              className="search"
            />

            {/* LIST */}
            <div className="list">
              {list.length === 0 ? (
                <p className="empty">Aucun rappel.</p>
              ) : (
                list.map((r) => {
                  const overdue = r.statut !== "LIVREE" && isOverdue(r.datePrevue);
                  return (
                    <div
                      key={r.id}
                      className={`erp-card ${overdue ? "danger" : r.statut === "EN_RETARD" ? "warning" : ""}`}
                      onClick={() => openItem(r)}
                    >
                      <div className="card-top">
                        <h3 className="card-title">
                          {r.type === "COMMANDE" ? `Commande #${r.reference}` : `Machine : ${r.reference}`}
                        </h3>
                        <span className="chip">{r.datePrevue}</span>
                      </div>

                      <p className="card-desc">{r.description}</p>

                      <div className="card-bottom">
                        <span className={`pill ${overdue ? "late" : r.statut.toLowerCase()}`}>
                          {overdue ? "⚠️ En retard" :
                           r.statut === "A_VENIR" ? "⏳ À venir" :
                           r.statut === "EN_RETARD" ? "⚠️ En retard" : "✅ Livrée"}
                        </span>

                        <span className={`pill prio ${r.priorite.toLowerCase()}`}>
                          {r.priorite === "URGENTE" ? "🔴 Urgente" :
                           r.priorite === "MOYENNE" ? "🟠 Moyenne" : "🟢 Basse"}
                        </span>
                      </div>
                    </div>
                  );
                })
              )}
            </div>
          </div>
        </div>

        {/* FAB ADD */}
        <IonFab vertical="bottom" horizontal="end" slot="fixed">
          <IonFabButton
            className="fab-outline"
            onClick={() => {
              resetAddForm(type); // ouvre formulaire selon onglet actuel
              setOpenAdd(true);
            }}
          >
            <IonIcon icon={addOutline} />
          </IonFabButton>
        </IonFab>

        {/* DETAILS MODAL */}
        <IonModal isOpen={openDetails} onDidDismiss={() => setOpenDetails(false)} className="date-modal">
          <IonContent className="date-modal-content">
            <div className="modal-head">
              <h3>Détails</h3>
              <IonButton fill="clear" onClick={() => setOpenDetails(false)}>
                <IonIcon icon={closeOutline} />
              </IonButton>
            </div>

            {selected && (
              <div className="details">
                <div className="details-row">
                  <span className="muted">Type</span>
                  <span>{selected.type}</span>
                </div>
                <div className="details-row">
                  <span className="muted">Référence</span>
                  <span>{selected.reference}</span>
                </div>

                {/* Champs spécifiques */}
                {selected.type === "COMMANDE" && selected.client && (
                  <div className="details-row">
                    <span className="muted">Client</span>
                    <span>{selected.client}</span>
                  </div>
                )}
                {selected.type === "MAINTENANCE" && selected.machine && (
                  <div className="details-row">
                    <span className="muted">Machine</span>
                    <span>{selected.machine}</span>
                  </div>
                )}

                <div className="details-row">
                  <span className="muted">Date prévue</span>
                  <span>{selected.datePrevue}</span>
                </div>
                <div className="details-row">
                  <span className="muted">Responsable</span>
                  <span>{selected.responsable}</span>
                </div>
                <div className="details-row">
                  <span className="muted">Priorité</span>
                  <span>{selected.priorite}</span>
                </div>
                <div className="details-row">
                  <span className="muted">Statut</span>
                  <span>{selected.statut}</span>
                </div>

                <div className="details-block">
                  <span className="muted">Description</span>
                  <p>{selected.description}</p>
                </div>
              </div>
            )}
          </IonContent>
        </IonModal>

        {/* ADD FORM MODAL */}
        <IonModal isOpen={openAdd} onDidDismiss={() => setOpenAdd(false)} className="date-modal">
          <IonContent className="date-modal-content">
            <div className="modal-head">
              <h3>Ajouter un rappel</h3>
              <IonButton fill="clear" onClick={() => setOpenAdd(false)}>
                <IonIcon icon={closeOutline} />
              </IonButton>
            </div>

            {/* TYPE */}
            <IonItem lines="none" className="form-item">
              <IonLabel>Type</IonLabel>
              <IonSelect
  interface="popover"
  value={newR.type}
  onIonChange={(e) => {
    const t = e.detail.value as RappelType;
    resetAddForm(t);
  }}
>
  <IonSelectOption value="COMMANDE">Commande</IonSelectOption>
  <IonSelectOption value="MAINTENANCE">Maintenance</IonSelectOption>
</IonSelect>

            </IonItem>

            {/* Champs spécifiques */}
            {newR.type === "COMMANDE" ? (
              <>
                <div className="field">
                  <label className="label">N° commande</label>
                  <IonInput className="input-plain"
                    value={newR.reference}
                    placeholder="CMD-1023"
                    onIonInput={(e) => setNewR(p => ({ ...p, reference: String(e.detail.value ?? "") }))}
                  />
                </div>
                <div className="field">
                  <label className="label">Client</label>
                  <IonInput className="input-plain"
                    value={newR.client || ""}
                    placeholder="Société X"
                    onIonInput={(e) => setNewR(p => ({ ...p, client: String(e.detail.value ?? "") }))}
                  />
                </div>
              </>
            ) : (
              <>
                <div className="field">
                  <label className="label">Machine</label>
                  <IonInput className="input-plain"
                    value={newR.reference}
                    placeholder="Tisser-03"
                    onIonInput={(e) => setNewR(p => ({ ...p, reference: String(e.detail.value ?? "") }))}
                  />
                </div>
                <IonItem lines="none" className="form-item">
                  <IonLabel>Type maintenance</IonLabel>
                  <IonSelect
  interface="popover"
  value={newR.maintenanceType || "PREVENTIVE"}
  onIonChange={(e) => setNewR(p => ({ ...p, maintenanceType: e.detail.value }))}
>
  <IonSelectOption value="PREVENTIVE">Préventive</IonSelectOption>
  <IonSelectOption value="CORRECTIVE">Corrective</IonSelectOption>
</IonSelect>

                </IonItem>
              </>
            )}

            <div className="field">
              <label className="label">Description</label>
              <IonInput className="input-plain"
                value={newR.description}
                placeholder={newR.type === "COMMANDE" ? "Livraison, production..." : "Maintenance, pièce, contrôle..."}
                onIonInput={(e) => setNewR(p => ({ ...p, description: String(e.detail.value ?? "") }))}
              />
            </div>

            <div className="field">
              <label className="label">Date prévue</label>
              <IonInput className="input-plain"
                type="date"
                value={newR.datePrevue}
                onIonInput={(e) => setNewR(p => ({ ...p, datePrevue: String(e.detail.value ?? "") }))}
              />
            </div>

            <IonItem lines="none" className="form-item">
              <IonLabel>Priorité</IonLabel>
             <IonSelect
  interface="popover"
  value={newR.priorite}
  onIonChange={(e) => setNewR(p => ({ ...p, priorite: e.detail.value }))}
>
  <IonSelectOption value="BASSE">Basse</IonSelectOption>
  <IonSelectOption value="MOYENNE">Moyenne</IonSelectOption>
  <IonSelectOption value="URGENTE">Urgente</IonSelectOption>
</IonSelect>

            </IonItem>

            <IonItem lines="none" className="form-item">
              <IonLabel>Statut</IonLabel>
            <IonSelect
  interface="popover"
  value={newR.statut}
  onIonChange={(e) => setNewR(p => ({ ...p, statut: e.detail.value }))}
>
  <IonSelectOption value="A_VENIR">À venir</IonSelectOption>
  <IonSelectOption value="EN_RETARD">En retard</IonSelectOption>
  <IonSelectOption value="LIVREE">Livrée</IonSelectOption>
</IonSelect>

            </IonItem>

            <div className="field">
              <label className="label">Responsable</label>
              <IonInput className="input-plain"
                value={newR.responsable}
                placeholder="Nom technicien / responsable"
                onIonInput={(e) => setNewR(p => ({ ...p, responsable: String(e.detail.value ?? "") }))}
              />
            </div>

            <div className="row">
              <IonButton className="go-btn" fill="outline" onClick={submitAdd}>
                Ajouter
              </IonButton>
            </div>
          </IonContent>
        </IonModal>

        <IonToast
          isOpen={toastOpen}
          message={toastMsg}
          duration={2400}
          position="top"
          onDidDismiss={() => setToastOpen(false)}
        />
        </div>
      </IonContent>
    </IonPage>
  );
};

export default Reminders;
