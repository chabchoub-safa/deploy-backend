import React, { useRef, useState, useEffect } from 'react';
import {
  IonPage,
  IonCard,
  IonCardHeader,
  IonCardTitle,
  IonCardContent,
  IonButton,
  IonIcon
} from '@ionic/react';
import { motion } from 'framer-motion';

import { Joystick } from 'react-joystick-component';
import mqtt from 'mqtt';
import { batteryCharging, wifi, power } from 'ionicons/icons';
import './VideoControl.css';

const VideoControl: React.FC = () => {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [mqttClient, setMqttClient] = useState<mqtt.MqttClient | null>(null);

  // États MQTT
  const [batteryLevel, setBatteryLevel] = useState('--%');
  const [networkStatus, setNetworkStatus] = useState('---');
  const [robotStatus, setRobotStatus] = useState('---');

  // Connexion MQTT
  useEffect(() => { 
    const client = mqtt.connect('wss://broker.emqx.io:8084/mqtt', {
      clientId: 'client_' + Math.random().toString(16).substring(2, 8),
      clean: true,
      connectTimeout: 4000,
      reconnectPeriod: 1000,
    });

    client.on('connect', () => {
      console.log('✅ MQTT connecté');
      client.subscribe('robot/status');
    });

    client.on('message', (topic, message) => {
      if (topic === 'robot/status') {
        const data = JSON.parse(message.toString());
        if (data.battery) setBatteryLevel(`${data.battery}%`);
        if (data.network) setNetworkStatus(data.network);
        if (data.state) setRobotStatus(data.state);
      }
    });

    setMqttClient(client);
   return () => {
    client.end();
  };
  }, []);

  // Joystick handlers
  const handleJoystickMove = (event: any) => {
    const direction = event.direction;
    console.log('Direction:', direction);
    mqttClient?.publish('RobotTest', direction);
  };

  const handleJoystickStop = () => {
    console.log('Joystick relâché');
    mqttClient?.publish('RobotTest', 'stop');
  };

  // Envoi commande capture vidéo
  const sendCaptureSignal = () => {
    if (mqttClient?.connected) {
      mqttClient.publish('RobotTest', 'capture');
      console.log('📤 Capture vidéo demandée');
    }
  };

  return (
    <IonPage>
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.6 }}
      ></motion.div>

      <IonCard className="video-control-card">
        <IonCardContent className="video-content-grid">
          {/* Titre */}
          <h2 className="top-title">🎥 Contrôle du Robot</h2>

          {/* Contenu en deux colonnes */}
          <div className="main-zone">
            {/* Flux vidéo */}
            <div className="video-frame">
              <video
                ref={videoRef}
                className="video-stream"
                autoPlay
                muted
                playsInline
                onCanPlay={() => videoRef.current?.play()}
              ></video>
            </div>

            {/* Paramètres robot */}
            <div className="robot-status">
              <p><IonIcon icon={batteryCharging} /> Batterie : {batteryLevel}</p>
              <p><IonIcon icon={wifi} /> Réseau : {networkStatus}</p>
              <p><IonIcon icon={power} /> Robot : {robotStatus}</p>
            </div>
          </div>

          {/* Bouton capture */}
          <IonButton className="skyblue-button" onClick={sendCaptureSignal}>
            Envoyer Capture
          </IonButton>

          {/* Joystick */}
          <div className="joystick-zone">
            <Joystick
              size={100}
              baseColor="#f0f0f0"
              stickColor="#007bff"
              move={handleJoystickMove}
              stop={handleJoystickStop}
            />
          </div>
        </IonCardContent>
      </IonCard>
    </IonPage>
  );
};

export default VideoControl;
